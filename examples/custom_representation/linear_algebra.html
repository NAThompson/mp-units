 18template<typename Rep>
 19std::ostream& operator<<(std::ostream& os, const ::vector<Rep>& v)
 20{
 21  os << "|";
 22  for (auto i = 0U; i < v.size(); ++i) {
 23    os << UNITS_STD_FMT::format(" {:>9}", v(i));
 24  }
 25  os << " |";
 26  return os;
 27}
 28
 29template<typename Rep>
 30std::ostream& operator<<(std::ostream& os, const ::matrix<Rep>& v)
 31{
 32  for (auto i = 0U; i < v.rows(); ++i) {
 33    os << "|";
 34    for (auto j = 0U; j < v.columns(); ++j) {
 35      os << UNITS_STD_FMT::format(" {:>9}", v(i, j));
 36    }
 37    os << (i != v.rows() - 1U ? " |\n" : " |");
 38  }
 39  return os;
 40}
